	   proximo: punt;
     Freg;

     { Se considera que la lista está  cargada }

	Sino I := 1;

		      I := I + 1;
		      T := *T.PROXIMO;
	     Fmientras;
	     Si (T = NIL) entonces
			Escribir ('ERROR, ', K, '> que elementos de la Lista')
		Sino Escribir ('VALOR: ', *T.VALOR);

3. Diseñar un algoritmo para insertar un nuevo elemento en el  k-ésimo lugar de una lista.
     - Si la lista está  vacía o si el valor de k está  fuera del rango del índice de la lista, invocar al procedimiento de ERROR.
     - En cualquier otro caso, efectuar el procedimiento de INSERCION.
---------------------------------------------------------------------------

Procedimiento Ej.4.1.insertar (K:entero, PRIMERO:punt, DATO:entero) es
AMBIENTE
Tipos
     punt= puntero a elto;
     elto= Registro
	   valor: entero;
	   proximo: puntero a elto;
     Freg;

Variables
     P, T, ANT: punt;
     I: entero;

ALGORITMO
     { Se considera que la lista está cargada }
     Si K = 1 entonces
		NUEVO(P);
	     *P.PROXIMO := PRIMERO;
	     PRIMERO	:= P;
	     *P.VALOR	:= DATO;
	Sino I := 1; ANT := NIL;
	     T := PRIMERO;
	     Mientras (T <> NIL) y (I <> K)  hacer
		      I := I + 1; ANT := T;
		      T := *T.PROXIMO;
	     Fmientras;
	     Si (I <> K) entonces
			Escribir ('ERROR, ', K, '> que eltos de la Lista')
		Sino NUEVO(P);
		     *P.PROXIMO  := *ANT.PROXIMO;  { o bien, T }
		     *ANT.PROXIMO := P;
		     *P.VALOR	:= DATO;
	     Fsi;
     Fsi;
